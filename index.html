gulp-buddha

My awesome gulp plugin

Install

$ npm install --save-dev gulp-buddha

Usage

var gulp = require('gulp');
var buddha = require('gulp-buddha');

gulp.task('default', function () {
    return gulp.src('src/file.ext')
        .pipe(buddha())
        .pipe(gulp.dest('dist'));
});

API

buddha(options)

options

foo

Type: boolean
Default: false
